\item[(1)] Use the given (or looked up) values to show the each 
constant in page 3 of Ryden:\\
Newtonian gravitational constant:\\
\hspace{2.0cm}
$G = 6.7\times10^{-11} \,m^3\,kg^{-1}\,s^{-2}$\\
Speed of light:\\
\hspace{2.0cm}
$c = 3.0\times10^8 \,m\,s^{-1}$\\
reduced Planck constant:\\
\hspace{2.0cm}
$\hbar = h/(2\pi) = 1.1\times10^{-34} \,J\,s \,(= 6.6\times10^{-16} \,eV)$\\

\begin{itemize}
  \vspace{-0.3cm}  
  \item[(a)] 1.1, Planck Length
    \[
l_p \equiv \left( \frac{G\hbar}{c^3} \right)^{1/2} = 1.6\times10^{-35} \,m
    \]  

  \vspace{-0.2cm}
  \item[(b)] 1.2, Planck Mass
    \[
M_p \equiv \left( \frac{\hbar c}{G} \right)^{1/2} = 2.2\times10^{-8} \,kg
    \]

  \vspace{-0.2cm}
  \item[(c)] 1.3, Planck Time
    \[
t_p \equiv \left( \frac{G\hbar}{c^5} \right)^{1/2} = 5.4\times10^{-44} \,s
    \]

  \vspace{-0.2cm}
  \item[(d)] 1.4, Planck Energy
    \[
E_p = M_p c^2  = 2.0\times10^9 \,J ( = 1.2\times10^{28} \,eV)
    \]

\end{itemize}

\item[(2)] And give meaning for those constants.\\
\end{itemize}
